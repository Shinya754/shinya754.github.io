class Solution:
    def longestCommonSubsequence(self, t1: str, t2: str) -> int:
        @lru_cache(None)
        def dfs(i, j):
            if not i or not j: return 0
            if t1[i-1] == t2[j-1]: return dfs(i-1, j-1) + 1
            return max(dfs(i-1,j), dfs(i, j-1))
        return dfs(len(t1), len(t2))
