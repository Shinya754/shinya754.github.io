public class BSTIterator {
    int index = 0;
    List<int> list;
    public BSTIterator(TreeNode root) {
        list = new List<int>();
        ReadTree(root);
    }
    public int Next() {
        return list[index++];
    }
    public bool HasNext() {
        if(index < list.Count){
            return true;
        }
        else{
            return false;
        }
    }
    public void ReadTree(TreeNode node){
        if(node.left!=null){
            ReadTree(node.left);
        }
        list.Add(node.val);
        if(node.right!=null){
            ReadTree(node.right);
        }
    }
}
