    int maxBit=30;
    Trie root=new Trie();
    public int FindMaximumXOR(int[] nums) {
        if(nums.Length==1)return 0;
        int ans=0;
        Add(nums[0]);
        for(int i=1;i<nums.Length;i++){
            ans=Math.Max(ans,Check(nums[i]));
            Add(nums[i]);
        }
        return ans;
    }
    ///å‘å­—å…¸æ ‘ä¸­æ·»åŠ ä¸€æ¡æ•°æ®
    public void Add(int num){
        Trie temp=root;

            if(((num>>i)&1)==1){
                if(temp.left==null)
                    temp.left=new Trie();
                temp=temp.left;
            }
            else{
                if(temp.right==null)
                    temp.right=new Trie();
                temp=temp.right;
            }
        }
    }

                }

                }
            }
        }

    }
