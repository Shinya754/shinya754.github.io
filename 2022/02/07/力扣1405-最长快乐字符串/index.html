class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        vector<pair<char,int>> vec;
        vec.push_back(make_pair('a',a));
        vec.push_back(make_pair('b',b));
        vec.push_back(make_pair('c',c));
        sort(vec.begin(),vec.end(),[](pair<char,int> p1,pair<char,int> p2){
            return p1.second>=p2.second;
        });
        if(vec[0].second==1 && vec[1].second==0){
            return string(1,vec[0].first);
        }
        string ans="";
        while(1){
            sort(vec.begin(),vec.end(),[](pair<char,int> p1,pair<char,int> p2){
                return p1.second>=p2.second;
            });
            for(int i=0;i<3;i++){
                int siz=ans.size();
                if(vec[i].second<=0){
                    break;
                }
                if(siz>=2 && ans[siz-2]==vec[i].first && ans[siz-1]==vec[i].first){
                    continue;
                }
                ans.push_back(vec[i].first);
                vec[i].second--;
                break;
            }
            int siz=ans.size();
            if(siz>=2 && ans[siz-2]==vec[0].first && ans[siz-1]==vec[0].first && vec[1].second<=0){
                break;
            }
            else if(vec[0].second==0 && vec[1].second==0 && vec[2].second==0){
                break;
            }
        }
        return ans;
    }
};
